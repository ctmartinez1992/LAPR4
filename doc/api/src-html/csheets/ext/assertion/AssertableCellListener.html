023    package csheets.ext.assertion;
024    
025    import java.util.EventListener;
026    
027    /**
028     * A listener interface for receiving notification on events occurring in an
029     * assertable cell.
030     * @author Einar Pehrson
031     */
032    public interface AssertableCellListener extends EventListener {
033    
034            /**
035             * Invoked when an assertion is added to or removed from a cell.
036             * @param cell the cell that was modified
037             */
038            public void assertionsChanged(AssertableCell cell);
039    }
