021    package csheets.ext.style.ui;
022    
023    import java.awt.Font;
024    import java.awt.event.KeyEvent;
025    
026    import javax.swing.ImageIcon;
027    
028    import csheets.ext.style.StylableCell;
029    import csheets.ext.style.StyleExtension;
030    import csheets.ui.ctrl.UIController;
031    
032    /**
033     * An italicity changing operation.
034     * @author Einar Pehrson
035     */
036    @SuppressWarnings("serial")
037    public class ItalicAction extends StyleAction {
038    
039            /**
040             * Creates a new italic action.
041             * @param uiController the user interface controller
042             */
043            public ItalicAction(UIController uiController) {
044                    super(uiController);
045            }
046    
047            protected String getName() {
048                    return "Italic";
049            }
050    
051            protected void defineProperties() {
052                    putValue(MNEMONIC_KEY, KeyEvent.VK_I);
053                    putValue(SMALL_ICON, new ImageIcon(StyleExtension.class.getResource("res/img/font_italic.gif")));
054            }
055    
056            /**
057             * Toggles the italicity of the selected cells in the focus owner table.
058             * @param cell the cell to which style should be applied
059             */
060            protected void applyStyle(StylableCell cell) {
061                    cell.setFont(new Font(cell.getFont().getFamily(),
062                            cell.getFont().getStyle() ^ Font.ITALIC, cell.getFont().getSize()));
063            }
064    }
