021    package csheets.ui.ctrl;
022    
023    import java.util.EventObject;
024    
025    import csheets.core.Workbook;
026    
027    /**
028     * An edit event is used to notify interested parties that a workbook has been
029     * modified.
030     * @author Einar Pehrson
031     */
032    @SuppressWarnings("serial")
033    public class EditEvent extends EventObject {
034    
035            /** The workbook that was modified */
036            private Workbook workbook;
037    
038            /**
039             * Creates a new edit event.
040             * @param source the source of the event
041             * @param workbook the workbook that was modified
042             */
043            public EditEvent(Object source, Workbook workbook) {
044                    super(source);
045    
046                    // Stores members
047                    this.workbook = workbook;
048            }
049    
050            /**
051             * Returns the active workbook.
052             * @return the active workbook
053             */
054            public Workbook getWorkbook() {
055                    return workbook;
056            }
057    }
