021    package csheets.ui.ctrl;
022    
023    import java.util.EventListener;
024    
025    /**
026     * A listener for receiving notification changes when a new active workbook,
027     * spreadsheet and/or cell is selected.
028     * @author Einar Pehrson
029     */
030    public interface SelectionListener extends EventListener {
031    
032            /**
033             * Invoked when the active workbook, spreadsheet and/or cell of the
034             * application is changed.
035             * @param event the selection event that was fired
036             */
037            public void selectionChanged(SelectionEvent event);
038    }
