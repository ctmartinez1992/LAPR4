021    package csheets.ui.ctrl;
022    
023    import java.awt.event.ActionEvent;
024    
025    import javax.swing.ImageIcon;
026    
027    import csheets.CleanSheets;
028    
029    /**
030     * A column insertion operation.
031     * @author Einar Pehrson
032     */
033    @SuppressWarnings("serial")
034    public class InsertColumnAction extends FocusOwnerAction {
035    
036            /**
037             * Creates a new column insertion action.
038             */
039            public InsertColumnAction() {}
040    
041            protected String getName() {
042                    return "Insert Column";
043            }
044    
045            protected void defineProperties() {
046                    setEnabled(false);
047                    putValue(SMALL_ICON, new ImageIcon(CleanSheets.class.getResource("res/img/insert_column.gif")));
048            }
049    
050            /**
051             * Inserts a column before the active cell in the focus owner table.
052             * @param event the event that was fired
053             */
054            public void actionPerformed(ActionEvent event) {
055                    if (focusOwner == null)
056                            return;
057            }
058    }
