021    package csheets.ui.ctrl;
022    
023    import java.awt.event.ActionEvent;
024    
025    import javax.swing.Action;
026    
027    /**
028     * A clipboard action, i.e. a cut, copy or paste operation.
029     * @author Einar Pehrson
030     */
031    public abstract class ClipboardAction extends FocusOwnerAction {
032    
033            /**
034             * Creates a new clipboard action.
035             */
036            public ClipboardAction() {}
037    
038            /**
039             * Forwards the event to the focus owner.
040             * @param event the event that was fired
041             */
042            public void actionPerformed(ActionEvent event) {
043                    if (focusOwner == null)
044                            return;
045                    String command = (String)event.getActionCommand();
046                    Action action = focusOwner.getActionMap().get(command);
047                    if (action != null)
048                            action.actionPerformed(
049                                    new ActionEvent(focusOwner, ActionEvent.ACTION_PERFORMED, null));
050            }
051    }
